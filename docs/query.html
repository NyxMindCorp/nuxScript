📊 Query System

Sistema de queries SQL-like para dados - como LINQ em C# ou Spark

Sintaxe

let users = [
    {name: "Alice", age: 25, city: "NYC"},
    {name: "Bob", age: 30, city: "LA"}
]

// Sintaxe de query
let young = select u in users where u.age < 30

Exemplos

// Query completo
let result = users
    |> where(.age >= 18)
    |> order_by("name")
    |> select("name", "city")

// Group by
let by_city = group_by(users, "city")
for city in by_city
    print("{city.key}: {city.count} users")
end

// Aggregates
let stats = users
    |> count()           // 2
    |> sum("age")        // 55
    |> avg("age")        // 27.5
    |> min("age")        // 25
    |> max("age")        // 30
