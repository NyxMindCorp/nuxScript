🏷️ Tipo System Avançado

Refinements, Brands, Newtypes, Opaque types

Refinements

// Tipos com validação
pos()              // número positivo
neg()              // número negativo  
nonempty()         // string não-vazia
email()            // email válido
uuid()             // UUID válido
url()              // URL válida
min(10)            // mínimo 10
max(100)           // máximo 100
length(1, 50)      // comprimento 1-50
in_range(0, 100)    //no range 0-100

// Usar
let positive = pos()
positive.check(5)    // true
positive.check(-1)  // false

Brands (Branded Types)

// Criar brand
let UserId = brand("UserId")

// Usar
let user_id = UserId.wrap(123)
// user_id.__brand__ === "UserId"
UserId.unwrap(user_id)  // 123

Newtypes

// Criar newtype
let Age = newtype("Age", struct_type("Number"))

let age = Age.wrap(25)
// age.__newtype__ === "Age"
Age.unwrap(age)  // 25

Opaque Types

// Tipo opaco (privado)
let Connection = opaque("Connection")

// wrap retorna getter privado
let conn = Connection.wrap(internal_data)
Connection.unwrap(conn)  // retorna dados

Type Providers

// Criar tipos
typeof(value)              // TypeProvider
struct_type("User")        // Struct type
enum_type("Status")        // Enum type
union_type(Int, String)     // Union type
optional_type(String)      // ?String
result_type(Ok, Err)        // Result type
array_of(Int)              // List[Int]
map_of(String, Int)       // Map[String, Int]
